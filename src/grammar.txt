<ASSIGNMENT-WORD> ::= <WORD> '=' <WORD>

<REDIRECTION> ::=  '>' <WORD>
                |  '<' <WORD>
                |  '>>' <WORD>
                |  '<<' <WORD>

<SIMPLE-COMMAND-ELEMENT> ::= <WORD>
                          |  <ASSIGNMENT-WORD>
                          |  <REDIRECTION>

<REDIRECTION-LIST> ::= <REDIRECTION> (<REDIRECTION>)*

<SIMPLE-COMMAND> ::=  <SIMPLE-COMMAND-ELEMENT> (<SIMPLE-COMMAND-ELEMENT>)*

<COMMAND> ::=  <SIMPLE-COMMAND>
            |  <SHELL-COMMAND>
            |  <SHELL-COMMAND> <REDIRECTION-LIST>

<SHELL-COMMAND> ::=  <SUBSHELL>

<SUBSHELL> ::=  '(' <COMMAND-GROUP> ')'
<COMMAND-GROUP> ::= <COMMAND-EXEC>
          | <COMMAND-EXEC> '&'
          | <COMMAND-EXEC> ';'

<COMMAND-EXEC> ::= <PIPELINE-COMMAND> '&&' <PIPELINE-COMMAND>
          | <PIPELINE-COMMAND> '||' <PIPELINE-COMMAND>
          | <PIPELINE-COMMAND> '&' <PIPELINE-COMMAND>
          | <PIPELINE-COMMAND> ';' <PIPELINE-COMMAND>
          | <PIPELINE-COMMAND>

<PIPELINE-COMMAND> ::= <PIPELINE>

<PIPELINE> ::= <PIPELINE> '|'  <PIPELINE>
       |  <COMMAND>

<WORD> ::= <ALPHA> | <WORD> <ALPHA> | <WORD> '_'

<WORD-LIST> ::= <WORD> (<WORD>)*
