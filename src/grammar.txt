<SHELL-COMMAND> ::=  <COMMAND-GROUP> | <SUBSHELL>

<COMMAND-GROUP> ::= <COMMAND-EXEC>
          | <COMMAND-EXEC> '&'
          | <COMMAND-EXEC> ';'

<COMMAND-EXEC> ::= <PIPELINE> '&&' <PIPELINE>
          | <PIPELINE> '||' <PIPELINE>
          | <PIPELINE> '&' <PIPELINE>
          | <PIPELINE> ';' <PIPELINE>
          | <PIPELINE>

<PIPELINE> ::= <COMMAND> ('|' <COMMAND>)*

<COMMAND> ::=  <SIMPLE-COMMAND>
            |  <SHELL-COMMAND>

<SIMPLE-COMMAND> ::=  <SIMPLE-COMMAND-ELEMENT> (<SIMPLE-COMMAND-ELEMENT>)*

<SIMPLE-COMMAND-ELEMENT> ::= <WORD>
                          |  <ASSIGNMENT-WORD>
                          |  <REDIRECTION>

<ASSIGNMENT-WORD> ::= <WORD> '=' <WORD>

<REDIRECTION> ::=  '>' <WORD>
                |  '<' <WORD>
                |  '>>' <WORD>
                |  '<<' <WORD>

<SUBSHELL> ::=  '(' <COMMAND-GROUP> ')'

<WORD> ::= <ALPHA> | <WORD> <ALPHA> | <WORD> '_'

<WORD-LIST> ::= <WORD> (<WORD>)*
